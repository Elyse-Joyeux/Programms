Strings in C

A C string is a null-terminated array of char. Understanding how they work in memory is essential.

What is a String?

C has no built-in string type. A string is simply a char array where the last element is the null character '\0' (ASCII 0). Every standard library function depends on this sentinel to know where the string ends.

⚠️A char array WITHOUT '\0' is NOT a string — using it with string functions causes undefined behavior.
Memory Layout

How "Hello" lives in memory (6 bytes, not 5):

'H'
[0]
'e'
[1]
'l'
[2]
'l'
[3]
'o'
[4]
'\0'
[5]

sizeof("Hello") = 6  ·  strlen("Hello") = 5

💡sizeof includes the null terminator. strlen counts only visible characters.
Declaration & Initialization
/* 1. Array — modifiable, size auto-calculated (=6) */
char name[] = "Alice";

/* 2. Array with explicit size — extra bytes zero-filled */
char name[50] = "Alice";

/* 3. Pointer to string literal — READ ONLY, do not modify! */
const char *name = "Alice";

/* 4. Manual character-by-character (must add '\0' yourself) */
char name[] = {'A','l','i','c','e','\0'};

/* 5. Uninitialized — must fill before use */
char buf[100];
🚫Modifying a string literal (char *p = "Hi"; p[0] = 'X';) is undefined behavior — use a char array instead.
Reading Strings
char buf[50];

/* scanf — stops at whitespace */
scanf("%49s", buf);

/* fgets — reads full line */
fgets(buf, sizeof(buf), stdin);

/* Strip newline from fgets */
buf[strcspn(buf, "\n")] = '\0';
🚫Never use gets() — no bounds check, removed in C11.
String Traversal
char str[] = "Hello";

/* Index-based */
for (int i = 0; str[i] != '\0'; i++)
    putchar(str[i]);

/* Pointer-based (idiomatic C) */
for (char *p = str; *p; p++)
    putchar(*p);

/* Reverse string */
int l=0, r=strlen(str)-1;
while(l < r) {
    char t=str[l]; str[l++]=str[r]; str[r--]=t;
}

Common Mistakes (Exam Traps)

❌ Wrong — comparing with ==

if (s1 == s2)  // compares addresses!

✅ Correct

if (strcmp(s1, s2) == 0)

❌ Wrong — overflow

char s[5] = "Hello"; // no '\0'!

✅ Correct

char s[6] = "Hello"; // len+1

❌ Wrong — modifying literal

char *p = "Hi"; p[0] = 'X'; // UB!

✅ Correct

char p[] = "Hi"; p[0] = 'X'; // OK

❌ Wrong — assignment

char s[10]; s = "Hello"; // error!

✅ Correct

strcpy(s, "Hello"); // use strcpy

Enumerations (enum)

Named integer constants that make code self-documenting and eliminate magic numbers.

What is an Enum?

An enum is a user-defined type whose values are symbolic names for integers. By default, values start at 0 and increment. Under the hood, they are just int.

/* Declaration */
enum Day { MON, TUE, WED, THU, FRI, SAT, SUN };
// MON=0  TUE=1  WED=2  THU=3  FRI=4  SAT=5  SUN=6

/* Usage */
enum Day today = WED;
printf("Day number: %d\n", today); // prints 2

/* typedef for cleaner usage (no 'enum' keyword) */
typedef enum { MON, TUE, WED, THU, FRI, SAT, SUN } Day;
Day today = WED;
Custom Values

You can assign specific integer values. Subsequent values continue incrementing from the last set value.

enum HTTPStatus {
    OK          = 200,
    CREATED     = 201,
    BAD_REQUEST = 400,
    UNAUTHORIZED= 401,
    NOT_FOUND   = 404,
    SERVER_ERROR= 500
};

enum Flags {
    READ    = 1,   // 001
    WRITE   = 2,   // 010
    EXECUTE = 4    // 100  (powers of 2 for bit flags)
};

enum Color { RED=5, GREEN, BLUE }; // GREEN=6, BLUE=7
💡Multiple enum constants CAN have the same value. Two names for the same number is valid C.
Enum in switch
enum Season { SPRING, SUMMER, AUTUMN, WINTER };

void describe(enum Season s) {
    switch(s) {
        case SPRING: puts("Warm"); break;
        case SUMMER: puts("Hot");  break;
        case AUTUMN: puts("Cool"); break;
        case WINTER: puts("Cold"); break;
    }
}
Enums as Bit Flags
enum Permission {
    NONE    = 0,
    READ    = 1 << 0,  // 1
    WRITE   = 1 << 1,  // 2
    EXECUTE = 1 << 2   // 4
};

int perm = READ | WRITE; // 3

if (perm & READ)
    puts("Can read");

perm |= EXECUTE;   // add flag
perm &= ~WRITE;    // remove flag

/* Enum in struct */
typedef struct {

    enum Day day;

Enum Traps (Exam Traps)
⚡Comparing enums from different types compiles fine in C — there's no type check. Always compare values, not names, since constants from different enums can share the same integer value.
/* Enum values are globally scoped — name conflicts! */
enum A { X = 1 };
enum B { X = 2 }; // ERROR: 'X' redefined

/* No automatic bounds checking */
enum Color { RED, GREEN, BLUE };
enum Color c = (enum Color)99; // compiles, UB territory

/* sizeof enum = sizeof int (typically) */
printf("%zu\n", sizeof(enum Day)); // usually 4

Unions

A union stores different data types in the same memory location — only one member is valid at a time.

What is a Union?

A union is like a struct where all members share the same memory. The size equals the largest member. Writing to one member overwrites all others.

union Data {
    int   i;    // 4 bytes
    float f;    // 4 bytes
    char  c;    // 1 byte
};              // sizeof(Data) = 4 (largest member)

union Data d;
d.i = 42;
printf("%d\n", d.i);   // OK: 42
d.f = 3.14f;
printf("%d\n", d.i);   // reading d.i now is UB!
🚫Reading from a union member that wasn't the last one written is undefined behavior (except for type-punning in certain compilers).
Union Memory Layout

All members start at the same address and overlap:

/* union Data — members overlap in memory */
/*
Address:  [0x100]  [0x101]  [0x102]  [0x103]
int i:   [   byte0    byte1    byte2    byte3   ]
float f: [   byte0    byte1    byte2    byte3   ]
char c:  [  byte0  ]
                  ^  all start at same address  ^
*/

union Mix { int i; double d; char s[20]; };
printf("%zu\n", sizeof(union Mix)); // 20 (char[20] is largest)
Tagged Union — The Right Way

Use a struct with an enum "tag" to track which member is currently valid:

typedef enum { INT_VAL, FLOAT_VAL, STR_VAL } ValueType;

typedef struct {
    ValueType type;   // tag: which member is active
    union {
        int   i;
        float f;
        char  s[64];
    } val;
} Value;

/* Writing */
Value v;
v.type = FLOAT_VAL;
v.val.f = 3.14f;

/* Reading safely */
switch(v.type) {
    case INT_VAL:   printf("int: %d\n",   v.val.i); break;
    case FLOAT_VAL: printf("float: %f\n", v.val.f); break;
    case STR_VAL:   printf("str: %s\n",   v.val.s); break;
}
✅Tagged unions are the standard safe pattern. Always track what the active member is via a separate tag field.
Union vs Struct Size
struct S { int a; float b; char c; };
union  U { int a; float b; char c; };

/* struct: members laid out sequentially
   + padding for alignment
   sizeof(S) = typically 12 */

/* union: all share same memory
   sizeof(U) = sizeof largest = 4 */
Type Punning
/* Inspect float bit pattern */
union FloatBits {
    float        f;
    unsigned int bits;
};

union FloatBits fb;
fb.f = 1.0f;
printf("%08X\n", fb.bits);
// 3F800000 (IEEE 754 for 1.0)

/* GCC allows this, C standard is UB.
   memcpy is the portable alternative. */
Union Initialization
union Data { int i; float f; char c; };

/* Only FIRST member can be initialized with {} */
union Data d1 = { 42 };       // d1.i = 42

/* C99 designated initializer — any member */
union Data d2 = { .f = 3.14f }; // d2.f = 3.14
union Data d3 = { .c = 'X' };   // d3.c = 'X'
💡Without designated initializers (C89), only the first member can be initialized at declaration.
Union Gotchas (Exam Traps)
GotchaExplanation
Size = largest memberNot the sum. sizeof always equals the largest member's size.
Only one active memberReading the wrong member is undefined behavior.
Endianness mattersType-punning via union exposes byte order — results differ on little vs big endian machines.
No constructor/destructorC unions do not call any initialization — members start uninitialized unless explicitly set.
Alignment paddingThe union size may be padded to satisfy the alignment of its largest member.
/* Alignment padding example */
union Padded {
    char   c;    // 1 byte
    double d;    // 8 bytes, alignment 8
};
// sizeof = 8 (padded for double's alignment)

Typedef Patterns
/* Struct typedef */
typedef struct { int x, y; } Point;
Point p = {3, 4};

/* Union typedef */
typedef union { int i; float f; } Number;
Number n; n.i = 10;

/* Enum typedef */
typedef enum { RED, GREEN, BLUE } Color;
Color c = GREEN;

/* Tagged union — combining all three */
typedef enum  { T_INT, T_FLOAT, T_STR }   Tag;
typedef union { int i; float f; char s[32]; } Payload;
typedef struct{ Tag tag; Payload data; }   Variant;
sizeof Quick Reference
/* Given: */
struct A { char c; int i; };          // typically 8 (4-byte alignment)
union  B { char c; int i; };          // 4 (= sizeof(int))
enum   C { X, Y, Z };               // typically 4 (= sizeof(int))
char      s[] = "Hi";               // 3 (includes '\0')
union  D { int i; char s[10]; };    // 12 (10 padded to 4-byte align)
⚡Struct and union sizes can include padding bytes. Always use sizeof — never manually add up member sizes.
Practical Pattern — State Machine with Enum
typedef enum { IDLE, RUNNING, PAUSED, STOPPED } State;

typedef struct {
    State   state;
    char    name[32];
    int     counter;
} Process;

void transition(Process *p, State next) {
    switch(p->state) {
        case IDLE:    if(next==RUNNING) p->state=RUNNING; break;
        case RUNNING: if(next==PAUSED||next==STOPPED) p->state=next; break;
        case PAUSED:  if(next==RUNNING||next==STOPPED) p->state=next; break;
        default: break;
    }
}
